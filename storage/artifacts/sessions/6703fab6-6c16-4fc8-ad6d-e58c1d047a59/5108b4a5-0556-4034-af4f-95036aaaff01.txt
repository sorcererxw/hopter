      snake = [

      ];
      direction = { x: 1, y: 0 };

    function spawnFood() {
      while (true) {
        const candidate = {
          x: Math.floor(Math.random() * tileCount),
          y: Math.floor(Math.random() * tileCount)
        };

        if (!snake.some((segment) => segment.x === candidate.x && segment.y === candidate.y)) {
          return candidate;
        }
      }
    }

    function updateScore() {
      scoreEl.textContent = String(score);
      bestScoreEl.textContent = String(bestScore);
    }

    function showOverlay(title, text, buttonText) {
      overlayTitleEl.textContent = title;
      overlayTextEl.textContent = text;
      restartButton.textContent = buttonText;
      overlayEl.classList.add("is-visible");
    }

      overlayEl.classList.remove("is-visible");
    }

    function setGameOver() {
      gameOver = true;
      statusTextEl.innerHTML = '<span class="game-over">Game over.</span> Press space or restart.';
      showOverlay("Game Over", `Final score: ${score}. Try to beat ${bestScore}.`, "Play Again");
    }

    function handleDirection(key) {
      const proposed = directions[key];
      if (!proposed || gameOver) {
        return;

      const reversing =
        snake.length > 1 &&

        proposed.y === -direction.y;

        started = true;
        statusTextEl.textContent = "Game in progress.";
        hideOverlay();
